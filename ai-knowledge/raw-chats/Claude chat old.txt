tsx"use client";

type Props = {
  sizes: number[];
  stockBySize: Record<number, number>;
  selected: number | null;
  onSelect: (size: number) => void;
};

export default function SizeSelector({ sizes, stockBySize, selected, onSelect }: Props) {
  return (
    <div>
      <p className="text-sm font-semibold text-gray-700 mb-2">Numara Seç</p>
      <div className="flex flex-wrap gap-2">
        {sizes.map((size) => {
          const stock = stockBySize[size] ?? 0;
          const isOut = stock === 0;
          const isSelected = selected === size;
          return (
            <button
              key={size}
              disabled={isOut}
              onClick={() => onSelect(size)}
              className={`
                w-12 h-12 rounded-lg text-sm font-semibold border-2 transition-all
                ${isOut ? "bg-gray-100 text-gray-400 border-gray-200 cursor-not-allowed line-through" : ""}
                ${!isOut && !isSelected ? "bg-white text-gray-800 border-gray-300 hover:border-orange-400" : ""}
                ${isSelected ? "bg-orange-500 text-white border-orange-500" : ""}
              `}
            >
              {size}
            </button>
          );
        })}
      </div>
    </div>
  );
}

16. app/product/[slug]/page.tsx — YENİ DOSYA
tsx"use client";
import { useState } from "react";
import { notFound } from "next/navigation";
import { getProductBySlug } from "@/lib/products";
import ProductGallery from "@/components/ProductGallery";
import SizeSelector from "@/components/SizeSelector";
import BuyRequestModal from "@/components/BuyRequestModal";

export default function ProductPage({ params }: { params: { slug: string } }) {
  const product = getProductBySlug(params.slug);
  if (!product) notFound();

  const [selectedSize, setSelectedSize] = useState<number | null>(null);
  const [modalOpen, setModalOpen] = useState(false);

  const totalStock = Object.values(product.stockBySize).reduce((a, b) => a + b, 0);
  const selectedStock = selectedSize ? product.stockBySize[selectedSize] : null;

  const handleBuyClick = () => {
    if (!selectedSize) {
      alert("Lütfen numara seçin.");
      return;
    }
    setModalOpen(true);
  };

  return (
    <>
      <div className="max-w-5xl mx-auto px-4 py-10">
        <div className="grid grid-cols-1 md:grid-cols-2 gap-10">
          {/* Gallery */}
          <ProductGallery images={product.images} name={product.name} />

          {/* Info */}
          <div className="flex flex-col gap-5">
            <div>
              <p className="text-sm text-orange-500 font-semibold uppercase tracking-wide">
                {product.category}
              </p>
              <h1 className="text-3xl font-extrabold text-gray-900 mt-1">{product.name}</h1>
            </div>

            <p className="text-3xl font-bold text-orange-600">
              {product.price.toLocaleString("tr-TR")} ₺
            </p>

            {/* Stock indicator */}
            {totalStock > 0 && totalStock <= 10 && (
              <p className="text-red-500 text-sm font-semibold">
                ⚠️ Son {totalStock} adet kaldı!
              </p>
            )}
            {totalStock === 0 && (
              <p className="text-red-600 text-sm font-semibold">Stok tükendi</p>
            )}

            <SizeSelector
              sizes={product.sizes}
              stockBySize={product.stockBySize}
              selected={selectedSize}
              onSelect={setSelectedSize}
            />

            {/* Size stock hint */}
            {selectedSize && selectedStock !== null && selectedStock > 0 && selectedStock <= 3 && (
              <p className="text-orange-500 text-sm">
                Bu numaradan sadece {selectedStock} adet kaldı!
              </p>
            )}

            <p className="text-gray-600 text-sm leading-relaxed">{product.description}</p>

            {/* Desktop buy button */}
            <button
              onClick={handleBuyClick}
              disabled={totalStock === 0}
              className="hidden md:block bg-orange-500 hover:bg-orange-600 disabled:bg-gray-300 text-white font-bold py-4 rounded-full text-lg transition-colors"
            >
              {totalStock === 0 ? "Stok Tükendi" : "Satın Almak İstiyorum"}
            </button>
          </div>
        </div>
      </div>

      {/* Mobile sticky bottom button */}
      <div className="md:hidden fixed bottom-0 left-0 right-0 bg-white border-t border-gray-200 p-4 z-40">
        <button
          onClick={handleBuyClick}
          disabled={totalStock === 0}
          className="w-full bg-orange-500 hover:bg-orange-600 disabled:bg-gray-300 text-white font-bold py-4 rounded-full text-lg transition-colors"
        >
          {totalStock === 0 ? "Stok Tükendi" : "Satın Almak İstiyorum"}
        </button>
      </div>

      {modalOpen && (
        <BuyRequestModal
          productName={product.name}
          selectedSize={selectedSize}
          onClose={() => setModalOpen(false)}
        />
      )}
    </>
  );
}
